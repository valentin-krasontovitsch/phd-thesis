        \raisebox\CdotAxis{%
          \scalebox{#1}{%
            \raisebox{-\CdotAxis}{%
              $\mathsurround=0pt #2#3$%
            }%
          }%
        }%
      }%
      % Remove depth that arises from scaling.
      \dp0=0pt %
      % Decrease scaled height.
      \sbox2{$#2\bullet$}%
      \ifdim\ht2<\ht0 %
        \ht0=\ht2 %
      \fi
      % Use the same width as the original \cdot.
      \sbox2{$\mathsurround=0pt #2#3$}%
      \hbox to \wd2{\hss\usebox{0}\hss}%
    }%
  }
